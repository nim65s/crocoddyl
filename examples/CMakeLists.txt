  talos_warm_start

MACRO(CHECK_LOG EXAMPLE MODULE)
  PYTHON_BUILD("${MODULE}" "${EXAMPLE}.py")
  ADD_PYTHON_UNIT_TEST("py-eg-${EXAMPLE}" "unittest/examples_logs.py" ".")
  SET(_ENV
    "CROCODDYL_EXAMPLE=${MODULE}/${EXAMPLE}"
    "PROJECT_SOURCE_DIR=${PROJECT_SOURCE_DIR}"
    "PYTHONPATH=${CMAKE_BINARY_DIR}"
    )
  SET_TESTS_PROPERTIES("py-eg-${EXAMPLE}" PROPERTIES ENVIRONMENT "${_ENV}")
ENDMACRO(CHECK_LOG EXAMPLE PATH)

FOREACH(EXAMPLE ${${PROJECT_NAME}_PYTHON_EXAMPLES})
  CHECK_LOG(${EXAMPLE} .)
ENDFOREACH(EXAMPLE ${${PROJECT_NAME}_EXAMPLES})

FOREACH(EXAMPLE ${${PROJECT_NAME}_PYTHON_EXAMPLES_LOCOMOTION})
  CHECK_LOG(${EXAMPLE} locomotion)
ENDFOREACH(EXAMPLE ${${PROJECT_NAME}_EXAMPLES_LOCOMOTION})
